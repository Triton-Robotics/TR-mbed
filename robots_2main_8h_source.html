143float BeybladeModulation(ChassisSpeeds cs){
144 float v = sqrt(cs.vX * cs.vX + cs.vY * cs.vY);
145 if(v < MAX_METERSPERSSECOND_60W){
146 return (MAX_METERSPERSSECOND_60W - v) * (MAX_RADIANSPERSSECOND_60W * 0.8);
147 }else{
148 return 0;
149 }
