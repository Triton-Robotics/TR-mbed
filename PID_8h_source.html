102struct lastfew {
103 long long last[lastX] = { 0 };
104 int arm = 0;
105 int lastY = 0;
106
107 lastfew(int lastY_ = 1) {
108 if(lastY_ > lastX){
109 lastY = lastX;
110 }else{
111 lastY = lastY_;
112 }
113 }
114
115 void add(long long l) {
116 last[arm] = l;
117 arm++;
118 if (arm == lastY)
119 arm = 0;
120 }
121
122 long long time() {
123 long long t = 0;
124 for (int i = 0; i < lastX; i++)
125 t += last[i];
126 return t / lastX;
127 }
128};
