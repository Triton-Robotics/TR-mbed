247 EIGEN_STRONG_INLINE void gemm_unrolled_MMA_iteration(
248  const DataMapper& res,
249  const Scalar* lhs_base,
250  const Scalar* rhs_base,
251  Index depth,
252  Index strideA,
253  Index offsetA,
254  Index& row,
255  Index col,
256  const Packet& pAlpha)
257 {
258  const Scalar* rhs_ptr = rhs_base;
