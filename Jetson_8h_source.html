84
85 int write_data_to_buff(Jetson::WriteState &write_state, char *buff,
86 int buff_size) {
87
88 // buff must be able to hold payload + header + checksum
89 if (buff_size < payload_size_ + 2) {
90 return -1;
91 }
92
93 // write payload into buff, leaving 1 byte for the header
94 get_packed_payload(write_state, &buff[1]);
95 buff[0] = header_;
96
97 uint8_t checksum = Jetson::calculateLRC(&buff[1], payload_size_);
98 buff[payload_size_ + 1] = checksum;
99
100 // payload + 1 header byte + 1 checksum byte
101 return payload_size_ + 2;
102 }
103
104 private:

106 char *buff) = 0;
107};

152 public:
153 const char header_;
154 const int payload_size_;
155 ReadPacket(char header, int payload_size)
156 : header_(header), payload_size_{payload_size} {};
157
158 virtual ~ReadPacket() = default;
159
160 // returns the amount of bytes consumed or -1 if invalid
161 int parse_buff(char *buff, int buff_size, Jetson::ReadState &read_state) {
162 if (buff[0] != header_) {
163 return -1;
164 }
165 if (buff_size < payload_size_ + 2) {
166 return -1;
167 }
168
169 char checksum = buff[payload_size_ + 1];
170 // calculate checksum without header
171 char calculated_checksum =
172 Jetson::calculateLRC(&buff[1], payload_size_);
173
174 if (checksum != calculated_checksum) {
175 return -1;
176 }
177
178 extract_payload(&buff[1], read_state);
179
180 // consumed header + payload + checksum bytes
181 return payload_size_ + 2;
182 }
183
184 private:
185 virtual void extract_payload(char *buff, Jetson::ReadState &read_state) = 0;
186};
